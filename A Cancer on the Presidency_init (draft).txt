        if (el.innerText.startsWith("Name:")) {
            const name = el.innerText.replace("Name:", "").trim();
            el.innerHTML = `<span class="typewriter-name">${name}</span>`;
        }
    });

            el.style.color = "#ff0000";
        }
    });
    const progressH3 = document.querySelector('#progress_bar h3');
    if (progressH3 && !progressH3.dataset.animated) {
        progressH3.dataset.animated = 'true';
        progressH3.classList.add('typewriter-name');

 if (!document.body.dataset.candMusicHooked) {
        document.body.dataset.candMusicHooked = 'true';
        setInterval(() => {
            const candSelect = document.getElementById('candidate_id');
            if (candSelect && candSelect.value === '10701' && !candSelect.dataset.played) {
                candSelect.dataset.played = 'true';
                if (window.playlist && window.loadAndPlay) {
                    window.playlist.currentSongIndex = 0;
                    window.loadAndPlay();
                }

function replaceRadioButtons() {
    const form = document.querySelector('form[name="question"]');
    if (!form) return;
    // Only run once per question render
    if (form.dataset.checkboxified) return;
    form.dataset.checkboxified = 'true';

    const radios = form.querySelectorAll('input[type="radio"]');
    if (!radios.length) return;

    radios.forEach(radio => {
    const label = document.querySelector('label[for="' + radio.id + '"]');
    const labelText = label ? label.innerHTML.trim() : '';

    const row = document.createElement('div');
    row.className = 'game_answer_row';

    const checkLabel = document.createElement('label');
    checkLabel.className = 'checkbox path';

    const checkbox = document.createElement('input');
    checkbox.type = 'checkbox';
    checkbox.name = 'game_answers';
    checkbox.value = radio.value;

    const svg = document.createElementNS('http://www.w3.org/2000/svg', 'svg');
    svg.setAttribute('viewBox', '0 0 21 21');
    const path = document.createElementNS('http://www.w3.org/2000/svg', 'path');
    path.setAttribute('d', 'M5,10.75 L8.5,14.25 L19.4,2.3 C18.8333333,1.43333333 18.0333333,1 17,1 L4,1 C2.35,1 1,2.35 1,4 L1,17 C1,18.65 2.35,20 4,20 L17,20 C18.65,20 20,18.65 20,17 L20,7.99769186');
    svg.appendChild(path);
    checkLabel.appendChild(checkbox);
    checkLabel.appendChild(svg);

    const textSpan = document.createElement('span');
    textSpan.innerHTML = labelText;
    row.appendChild(checkLabel);
    row.appendChild(textSpan);

    checkbox.addEventListener('change', () => {
        if (checkbox.checked) {
            form.querySelectorAll('input[type="checkbox"]').forEach(cb => {
                if (cb !== checkbox) cb.checked = false;
            });
        }
    });

    row.addEventListener('click', (e) => {
        if (e.target !== checkbox) checkbox.click();
    });

    // Insert row, remove radio, label and <br>
    radio.insertAdjacentElement('beforebegin', row);
    if (label) label.remove();
    const br = radio.nextElementSibling;
    if (br && br.tagName === 'BR') br.remove();
    radio.remove();
});

    // Patch the CONTINUE button to read checkbox value back into engine
    const continueBtn = document.getElementById('answer_select_button');
    if (continueBtn) {
        continueBtn.addEventListener('click', () => {
            const checked = form.querySelector('input[type="checkbox"]:checked');
            if (checked) {
                let hidden = form.querySelector('input[type="radio"].checkbox-proxy');
                if (!hidden) {
                    hidden = document.createElement('input');
                    hidden.type = 'radio';
                    hidden.name = 'game_answers';
                    hidden.className = 'checkbox-proxy';
                    hidden.style.display = 'none';
                    form.appendChild(hidden);
                }
                hidden.value = checked.value;
                hidden.checked = true;
            }
        }, true); // capture phase so it fires before engine handler
    }
}

function makeDraggable(elId, savedPos) {
    const el = document.getElementById(elId);
    if (!el || el.dataset.dragHooked) return;
    el.dataset.dragHooked = 'true';

    el.style.left = savedPos.left;
    el.style.top = savedPos.top;
    el.style.cursor = 'grab';

    // create pushpin
    const pin = document.createElement('div');
    pin.className = 'map-pin';
    pin.innerHTML = `<svg xmlns='http://www.w3.org/2000/svg' width='20' height='28' viewBox='0 0 20 28'><circle cx='10' cy='8' r='7' fill='#cc2200' stroke='#881100' stroke-width='1.5'/><ellipse cx='10' cy='7' rx='3' ry='2.5' fill='#ff6644' opacity='0.5'/><rect x='9' y='14' width='2' height='12' rx='1' fill='#888'/></svg>`;
    const parent = document.getElementById('main_content_area') || el.parentElement;
    parent.appendChild(el);
    parent.appendChild(pin);
    pin.style.left = (parseInt(el.style.left) + el.offsetWidth / 2 - 10) + 'px';
    pin.style.top = (parseInt(el.style.top) - 20) + 'px';

    function updatePin() {
        pin.style.left = (el.offsetLeft + el.offsetWidth / 2 - 10) + 'px';
        pin.style.top = (el.offsetTop - 20) + 'px';
    }

    let isDrag = false, startX, startY, startLeft, startTop;

    el.addEventListener('mousedown', (e) => {
        if (e.target.tagName === 'BUTTON') return;
        isDrag = true;
        startX = e.clientX;
        startY = e.clientY;
        startLeft = el.offsetLeft;
        startTop = el.offsetTop;
        el.style.cursor = 'grabbing';
        e.preventDefault();
    });

    document.addEventListener('mousemove', (e) => {
        if (!isDrag) return;
        el.style.left = (startLeft + e.clientX - startX) + 'px';
        el.style.top = (startTop + e.clientY - startY) + 'px';
        updatePin();
    });

    document.addEventListener('mouseup', () => {
        if (!isDrag) return;
        isDrag = false;
        el.style.cursor = 'grab';
        savedPos.left = el.style.left;
        savedPos.top = el.style.top;
        updatePin();
    });

    new MutationObserver(updatePin).observe(el, { attributes: true, attributeFilter: ['style'] });
}

class Song {
    constructor(title, artist, coverLink, audioLink) {
        this.title = title;
        this.artist = artist;
        this.coverLink = coverLink;
        this.audioLink = audioLink;
    }
    getTitle()     { return this.title; }
    getArtist()    { return this.artist; }
    getCoverLink() { return this.coverLink; }
    getAudioLink() { return this.audioLink; }
}

class Playlist {
    constructor() {
        this.songs = [];
        this.currentSongIndex = 0;
    }
    addSong(song) { this.songs.push(song); }
    getCurrentSong() { return this.songs[this.currentSongIndex]; }
    playNext() { this.currentSongIndex = (this.currentSongIndex + 1) % this.songs.length; }
    playPrevious() { this.currentSongIndex = (this.currentSongIndex - 1 + this.songs.length) % this.songs.length; }
}
window.Playlist = Playlist;
window.Song = Song;

function updateUI() {
    const currentSong = playlist.getCurrentSong();
    const titleEl = document.getElementById('reel-track-title');
    const artistEl = document.getElementById('reel-track-artist');
    if (titleEl) {
        titleEl.textContent = currentSong.getTitle();
        titleEl.title = currentSong.getTitle();
    }
    if (artistEl) {
        artistEl.textContent = currentSong.getArtist();
        artistEl.title = currentSong.getArtist();
    }
}

    // Reel animation helpers
    const leftReel = document.getElementById("leftReel");

    function setReelAnimation(state) {
        // state: 'play', 'rw', 'ff', 'pause'
        [leftReel, rightReel].forEach(r => {
            r.className = "";
        });
        if (state === 'play') {
            leftReel.className = "reverse normal";
            rightReel.className = "reverse normal";
        } else if (state === 'ff') {
            leftReel.className = "reverse fast";
            rightReel.className = "reverse fast";
        } else if (state === 'rw') {
            leftReel.className = "forward fast";
            rightReel.className = "forward fast";
        }
        // pause: no class = stopped
    }

    // Tape strip + tape resize
    function updateTapeVisuals() {
    if (!audio.duration || isNaN(audio.duration)) return;
    const pct = audio.currentTime / audio.duration * 100;

    const maxTape = 210, minTape = 10;
    const tapeRange = maxTape - minTape;
    const leftSize = maxTape - (tapeRange * pct / 100);
    const rightSize = minTape + (tapeRange * pct / 100);

    // Fixed centre points for each tape
    const leftCX = 125, leftCY = 127;
    const rightCX = 378, rightCY = 127;

    leftTape.style.width = leftSize + "px";
    leftTape.style.height = leftSize + "px";
    leftTape.style.left = (leftCX - leftSize / 2) + "px";
    leftTape.style.top = (leftCY - leftSize / 2) + "px";

    rightTape.style.width = rightSize + "px";
    rightTape.style.height = rightSize + "px";
    rightTape.style.left = (rightCX - rightSize / 2) + "px";
    rightTape.style.top = (rightCY - rightSize / 2) + "px";

    // Strip rotation
    const leftLimits = [7, 2];
    const rightLimits = [-20, 3];
    const leftAngle = leftLimits[0] + (leftLimits[1] - leftLimits[0]) / 100 * pct;
    const rightAngle = rightLimits[0] + (rightLimits[1] - rightLimits[0]) / 100 * pct;
    leftStrip.style.transform = `rotate(${leftAngle}deg)`;
    rightStrip.style.transform = `rotate(${rightAngle}deg)`;
    }

    let visualTimer = null;
    function startVisuals(state) {
        if (visualTimer) clearInterval(visualTimer);
        setReelAnimation(state);
        visualTimer = setInterval(updateTapeVisuals, 100);
    }
    function stopVisuals() {
        if (visualTimer) clearInterval(visualTimer);
        visualTimer = null;
        setReelAnimation('pause');
    }

    // Playback state
    let playbackState = 'pause';
    let isPlaying = false;

  function loadAndPlay() {
        if (window._introSound) {
            window._introSound.pause();
            window._introSound.currentTime = 0;
        }
        audio.src = playlist.getCurrentSong().getAudioLink();
        updateUI();
        audio.play();
        playbackState = 'play';
        playBtn.classList.add('active');
        startVisuals('play');
        showStrips();
    }

   function showStrips() {
    leftStrip.style.height = '33px';
    rightStrip.style.height = '261px';
}

playBtn.addEventListener('click', () => {
        if (playbackState !== 'pause') {
            audio.pause();
            playbackState = 'pause';
            playBtn.classList.remove('active');
            stopVisuals();
        } else {
            audio.play();
            playbackState = 'play';
            playBtn.classList.add('active');

        setReelAnimation('rw');

            audio.currentTime = Math.max(0, audio.currentTime - 2);
            updateTapeVisuals();
        }, 100);
        rwBtn.addEventListener('mouseup', () => {
            clearInterval(rwInterval);
            if (isPlaying) {
                audio.play();
                playbackState = 'play';
                setReelAnimation('play');
            } else {
                playbackState = 'pause';
                setReelAnimation('pause');
            }
        }, { once: true });

        audio.pause();
        playbackState = 'ff';
        setReelAnimation('ff');
        const ffInterval = setInterval(() => {
            audio.currentTime = Math.min(audio.duration, audio.currentTime + 2);
            updateTapeVisuals();
        }, 100);
        ffBtn.addEventListener('mouseup', () => {
            clearInterval(ffInterval);
            if (isPlaying) {
                audio.play();
                playbackState = 'play';
                setReelAnimation('play');
            } else {
                playbackState = 'pause';
                setReelAnimation('pause');
            }
        }, { once: true });
    });

        audio.volume = Math.min(1, audio.volume + 0.15);

    function _positionRecorder() {
        const gameWindow = document.getElementById('game_window');

        const contentSingleRect = contentSingle.getBoundingClientRect();
        tapeRecorder.style.left = (gameWindowRect.left - contentSingleRect.left - 20) + 'px';
        tapeRecorder.style.top = (gameWindowRect.bottom - contentSingleRect.top + window.scrollY) + 'px';
    }

    setTimeout(() => {
        _positionRecorder();
        tapeRecorder.style.opacity = '1';
    }, 500);

    window.addEventListener('resize', () => {
        if (!_playerDragged) _positionRecorder();
    });

    tapeRecorder.addEventListener('mousedown', (e) => {
        if (e.target.tagName === 'BUTTON' || e.target.tagName === 'INPUT' || e.target.tagName === 'A' || e.target.tagName === 'SELECT') return;
        _playerDragged = true;
        isDragging = true;
        const recRect = tapeRecorder.getBoundingClientRect();
        dragOffsetX = e.clientX - recRect.left;
        dragOffsetY = e.clientY - recRect.top;
        tapeRecorder.style.cursor = 'grabbing';
        e.preventDefault();
    });
    window.loadAndPlay = loadAndPlay;

}
